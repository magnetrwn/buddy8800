167  inline void _ALU(cpu_registers8 src, u8 alu, bool is_immediate) {
168  u16 a = state.A();
169  u16 with = (is_immediate) ? fetch() : ((is_memref(src)) ? memory[state.HL()] : state.get_register8(src));
170  u16 result;
171 
172  switch (alu) {
173  case 0b000:
174  case 0b001:
175  result = a + with + ((alu == 0b001) ? state.flgC() : 0);
176  state.flgAC(((a & 0x0F) + (with & 0x0F) + ((alu == 0b001) ? state.flgC() : 0)) > 0x0F);
177  break;
178  case 0b010:
179  case 0b011:
180  result = a - with - ((alu == 0b011) ? state.flgC() : 0);
181  state.flgAC((a & 0x0F) >= (with & 0x0F));
182  break;
183  case 0b100:
184  result = a & with;
185  state.flgAC((a | with) & 0x08);
186  break;
187  case 0b101:
188  case 0b110:
189  result = (alu == 0b101) ? (a ^ with) : (a | with);
190  state.flgAC(false);
191  break;
192  case 0b111:
193  result = a - with;
194  state.flgAC((a & 0x0F) >= (with & 0x0F));
195  state.flgC(result & 0x100);
196  state.set_Z_S_P_flags(result);
197  break;
198  default: throw std::runtime_error("Invalid ALU operation.");
199  }
200 
201  state.flgC(result & 0x100);
202  state.set_Z_S_P_flags(result);
203  if (alu != 0b111)
204  state.A(result);
205  }
